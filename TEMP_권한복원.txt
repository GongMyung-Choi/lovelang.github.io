@echo off
title 🔧 TEMP 및 AppData 권한 복원
color 0a

echo 🔍 TEMP 및 AppData 접근 권한 복원 중...

takeown /f "%TEMP%" /r /d y >nul 2>&1
icacls "%TEMP%" /grant %username%:F /T /C >nul 2>&1

takeown /f "%USERPROFILE%\AppData\Local\Hnc" /r /d y >nul 2>&1
icacls "%USERPROFILE%\AppData\Local\Hnc" /grant %username%:F /T /C >nul 2>&1

takeown /f "%APPDATA%\Hnc" /r /d y >nul 2>&1
icacls "%APPDATA%\Hnc" /grant %username%:F /T /C >nul 2>&1

echo ✅ 권한 복원 완료!
echo 💡 한글 다시 실행해보세요.
pause
exit
